1️⃣ Vérifier le Chargement du Script

// Vérifier si Flowise.js est chargé
const scripts = Array.from(document.querySelectorAll('script')).map(s => s.src);
const flowiseScript = scripts.find(s => s.includes('Flowise.js'));
console.log('📜 Script Flowise:', flowiseScript || '❌ NON TROUVÉ');

2️⃣ Scanner les Tables

// Scanner toutes les tables
const selector = 'table.min-w-full.border.border-gray-200.dark\\:border-gray-700.rounded-lg';
const tables = document.querySelectorAll(selector);
console.log('📊 Tables trouvées:', tables.length);

tables.forEach((table, i) => {
    const headers = Array.from(table.querySelectorAll('th')).map(th => th.textContent.trim());
    const hasFlowise = headers.some(h => h.toLowerCase() === 'flowise');
    console.log(`\nTable ${i+1}:`);
    console.log('  Headers:', headers);
    console.log('  Flowise:', hasFlowise ? '✅ OUI' : '❌ NON');
    console.log('  Processed:', table.classList.contains('n8n-processed') ? '✅ OUI' : '❌ NON');
});

3️⃣ Forcer le Rechargement du Script

// Forcer le rechargement de Flowise.js
const oldScript = document.querySelector('script[src*="Flowise.js"]');
if (oldScript) oldScript.remove();

const script = document.createElement('script');
script.src = '/Flowise.js?v=' + Date.now();
script.onload = () => {
    console.log('✅ Flowise.js rechargé avec succès');
    console.log('🔄 Attendez 2 secondes pour l\'initialisation...');
    setTimeout(() => {
        console.log('✅ Initialisation terminée');
        console.log('📝 Envoyez maintenant un message avec une table Flowise');
    }, 2000);
};
script.onerror = (e) => {
    console.error('❌ Erreur de chargement:', e);
};
document.head.appendChild(script);

4️⃣ Nettoyer le LocalStorage

// Nettoyer les notifications qui saturent le localStorage
localStorage.removeItem('clara-notifications');
localStorage.removeItem('notifications');
localStorage.removeItem('clara-notification-history');

let totalSize = 0;
for (let key in localStorage) {
    if (localStorage.hasOwnProperty(key)) {
        totalSize += new Blob([localStorage.getItem(key)]).size;
    }
}
console.log('✅ Notifications supprimées');
console.log('📊 Taille localStorage:', (totalSize / 1024).toFixed(2), 'KB');
console.log('🔄 Rechargez la page avec Ctrl+F5');
